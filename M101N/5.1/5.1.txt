db.posts.aggregate( 
  [
    {
      $project: {
        "_id": "$comments"
        }
    },
    {
      $unwind: "$_id"
    },
    {
      $group: {
        "_id": "$_id.author",
        "count": {
          $sum: 1
        }
      }
    },
    {
      $sort: {
        "count": 1
      }
    }
  ]
);
